Creating Holiday Variables

# Independence Day
d$july4 <- ifelse(format(d$date, "%m-%d") == "07-04",1,0)

# Juneteenth
d$juneteenth <- ifelse(format(d$date, "%m-%d") == "06-19",1,0)

# Mother's Day
mother <- as.Date(sapply(d$yr, function(y) {
  x <- as.Date(paste0(y, "-05-01"))
  x + (8 - as.POSIXlt(x)$wday) %% 7 + 6
}))
d$mother <- ifelse(d$date == mother,1,0)

# Father's Day
father <- as.Date(sapply(d$yr, function(y) {
  x <- as.Date(paste0(y, "-06-01"))
  x + (8 - as.POSIXlt(x)$wday) %% 7 + 13
}))
d$father <- ifelse(d$date == father,1,0)

# Memorial Day
memorial <- as.Date(sapply(d$yr, function(y) {
  x <- as.Date(paste0(y, "-05-01"))
  x - as.POSIXlt(x)$wday %% 7 + 29
}))
d$memorial <- ifelse(d$date == memorial,1,0)

# Labor Day
labor <- as.Date(sapply(d$yr, function(y) {
  x <- as.Date(paste0(y, "-09-01"))
  x + (8 - as.POSIXlt(x)$wday) %% 7
}))
d$labor <- ifelse(d$date == labor,1,0)

# single holiday
d$holiday <- as.integer(d$july4 | d$memorial | d$labor | d$juneteenth | d$mother | d$father)

Creating Lags 0-6

# Create lag0 through lag6 within each year
for (i in 0:6) {
  d[[paste0("lag", i)]] <- unlist(
    lapply(temp_by_year, function(x) {
      if (i == 0) {
        x
      } else {
        c(rep(NA, i), head(x, -i))
      }
    }),
    use.names = FALSE
  )
}

       main = names(res_list)[i])
  
  axis(1, at = lags)
  
  ## CI polygon
